def twod_array():

    alphabet_list = ["A","B","C","D","E","F","G","H"]


    for a in range(len(alphabet_list)):
        for i in range(1,9):
            print(" "+str(alphabet_list[a])+str(i), end="")
        print("\n")

twod_array()
